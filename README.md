# FriendsBirthdays
Web application to store birthdays of friends.

# Java Backend:
https://start.spring.io/ \
Spring initializr: \
  Project - Maven \
  Language - Java \
  Spring Boot - 3.1.1 \
  Packaging - jar \
  Java - 17 \
  Dependencies: Spring Web (Web), Spring Data JPA (SQL), MySQL Server (SQL)

# React Frontend:
  New React App: \
    npx create-react-app my-app \
    cd my-app \
    npm start
